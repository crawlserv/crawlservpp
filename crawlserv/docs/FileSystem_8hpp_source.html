205  inline std::vector<std::string> listFilesInPath(
206  std::string_view pathToDir,
207  std::string_view fileExtension
208  ) {
209  std::vector<std::string> result;
210 
211  // open path
212  const stdfs::path path(pathToDir);
213 
214  if(!stdfs::exists(path)) {
215  std::string exceptionString{"'"};
216 
217  exceptionString += pathToDir;
218  exceptionString += "' does not exist";
219 
220  throw Exception(exceptionString);
221  }
222 
223  if(!stdfs::is_directory(path)) {
224  std::string exceptionString{"'"};
225 
226  exceptionString += pathToDir;
227  exceptionString += "' is not a directory";
228 
229  throw Exception(exceptionString);
230  }
231 
232  // iterate through items
233  try {
234  for(const auto& it: stdfs::recursive_directory_iterator(path)) {
235  if(it.path().extension().string() == fileExtension) {
236  result.emplace_back(it.path().string());
237  }
238  }
239  }
240  catch(const stdfs::filesystem_error& e) {
241  std::string exceptionString{"Could not iterate over the files in '"};
242 
243  exceptionString += pathToDir;
244  exceptionString += "': ";
245  exceptionString += e.what();
246 
247  throw Exception(exceptionString);
248  }
249 
250  return result;
251  }

266  inline bool contains(std::string_view pathToDir, std::string_view pathToCheck) {
267  if(!FileSystem::isValidDirectory(pathToDir)) {
268  std::string exceptionString{"'"};
269 
270  exceptionString += pathToDir;
271  exceptionString += "' is not a valid directory";
272 
273  throw Exception(exceptionString);
274  }
275 
276  // make paths absolute
277  try {
278  const auto absPathToDir{
279  stdfs::canonical(pathToDir)
280  };
281 
282 #ifndef INCLUDE_STD_FILESYSTEM_EXPERIMENTAL
283  // create weakly canonical path
284  auto absPathToCheck{
285  stdfs::weakly_canonical(pathToCheck)
286  };
287 #else
288  // create approximate canonical path â no symlinks supported!
289  auto completePathToCheck{
290  stdfs::absolute(pathToCheck)
291  };
292 
293  stdfs::path absPathToCheck;
294 
295  for(const auto& segment : completePathToCheck) {
296  if(segment == "..") {
297  absPathToCheck = absPathToCheck.parent_path();
298  }
299  else if(segment == ".") {
300  continue;
301  }
302  else {
303  absPathToCheck /= segment;
304  }
305 
306  if(stdfs::is_symlink(absPathToCheck)) {
307  std::string exceptionString{
308  "Path contains unsupported symlink: '"
309  };
310 
311  exceptionString += absPathToCheck;
312  exceptionString += "'";
313 
314  throw Exception(exceptionString);
315  }
316  }
317 #endif
318 
319  // remove filename if necessary
320  if(absPathToCheck.has_filename()) {
321  absPathToCheck.remove_filename();
322  }
323 
324  // compare number of path components
325  if(
326  std::distance(absPathToDir.begin(), absPathToDir.end())
327  > std::distance(absPathToCheck.begin(), absPathToCheck.end())
328  ) {
329  // pathToCheck cannot be contained in a path with more components
330  return false;
331  }
332 
333  // compare as many components as there are in pathToDir
334  return std::equal(absPathToDir.begin(), absPathToDir.end(), absPathToCheck.begin());
335  }
336  catch(const stdfs::filesystem_error& e) {
337  std::string exceptionString{
338  "Could not make paths absolute: '"
339  };
340 
341  exceptionString += pathToDir;
342  exceptionString += "' and '";
343  exceptionString += pathToCheck;
344  exceptionString += "': ";
345  exceptionString += e.what();
346 
347  throw Exception(exceptionString);
348  }
349  }
