268  inline void Config::checkOptions() {
269  // check corpus chunk size (in percent of the maximum packet size allowed by the MySQL server)
270  if(
271  this->config.generalCorpusSlicing < minPercentageCorpusSlices
272  || this->config.generalCorpusSlicing > maxPercentageCorpusSlices
273  ) {
274  this->config.generalCorpusSlicing = defaultPercentageCorpusSlices;
275 
276  this->warning(
277  "Invalid corpus chunk size reset to "
278  + std::to_string(defaultPercentageCorpusSlices)
279  + "% of the maximum packet size allowed by the MySQL server."
280  );
281  }
282 
283  // check properties of input fields
284  const auto completeInputs{
285  std::min({ // number of complete inputs (= min. size of all arrays)
286  this->config.generalInputFields.size(),
287  this->config.generalInputSources.size(),
288  this->config.generalInputTables.size()
289  })
290  };
291 
292  bool incompleteInputs{false};
293 
294  // remove field names that are not used
295  if(this->config.generalInputFields.size() > completeInputs) {
296  this->config.generalInputFields.resize(completeInputs);
297 
298  incompleteInputs = true;
299  }
300 
301  // remove field sources that are not used
302  if(this->config.generalInputSources.size() > completeInputs) {
303  this->config.generalInputSources.resize(completeInputs);
304 
305  incompleteInputs = true;
306  }
307 
308  // remove field source tables that are not used
309  if(this->config.generalInputTables.size() > completeInputs) {
310  // remove sources of incomplete datetime queries
311  this->config.generalInputTables.resize(completeInputs);
312 
313  incompleteInputs = true;
314  }
315 
316  // warn about incomplete input fields
317  if(incompleteInputs) {
318  this->warning(
319  "\'input.fields\', \'.sources\' and \'.tables\'"
320  " should have the same number of elements."
321  );
322 
323  this->warning("Incomplete input field(s) removed from configuration.");
324  }
325 
326  // check algo options
327  this->checkAlgoOptions();
328  }
