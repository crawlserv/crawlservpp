637  inline void Config::checkOptions() {
638  // check for link extraction query
639  if(this->config.crawlerQueriesLinks.empty()) {
640  throw Exception(
641  "Crawler::Config::checkOptions():"
642  " No link extraction query has been specified"
643  );
644  }
645 
646  // check number of URLs to crawl at once
647  if(this->config.crawlerUrlChunks == 0) {
648  this->config.crawlerUrlChunks = defaultUrlChunks;
649 
650  this->warning(
651  "Invalid value for \'ur.chunks\' ignored (was zero),"
652  "default value used"
653  );
654  }
655 
656  // check properties of archives
657  bool incompleteArchives{false};
658 
659  const auto completeArchives{
660  std::min({ // number of complete archives (= min. size of all arrays)
661  this->config.crawlerArchivesNames.size(),
662  this->config.crawlerArchivesUrlsMemento.size(),
663  this->config.crawlerArchivesUrlsTimemap.size()
664  })
665  };
666 
667  // remove names that are not used
668  if(this->config.crawlerArchivesNames.size() > completeArchives) {
669  this->config.crawlerArchivesNames.resize(completeArchives);
670 
671  incompleteArchives = true;
672  }
673 
674  // remove memento URL templates that are not used
675  if(this->config.crawlerArchivesUrlsMemento.size() > completeArchives) {
676  this->config.crawlerArchivesUrlsMemento.resize(completeArchives);
677 
678  incompleteArchives = true;
679  }
680 
681  // remove timemap URL templates that are not used
682  if(this->config.crawlerArchivesUrlsTimemap.size() > completeArchives) {
683  this->config.crawlerArchivesUrlsTimemap.resize(completeArchives);
684 
685  incompleteArchives = true;
686  }
687 
688  // warn about incomplete archives
689  if(incompleteArchives) {
690  this->warning(
691  "\'archives.names\', \'.urls.memento\' and \'.urls.timemap\'"
692  " should have the same number of elements."
693  );
694 
695  this->warning("Incomplete archive(s) removed from configuration.");
696  }
697 
698  // check properties of counters
699  bool incompleteCounters{false};
700 
701  const auto completeCounters{
702  std::min({ // number of complete counters (= min. size of arrays)
703  this->config.customCounters.size(),
704  this->config.customCountersStart.size(),
705  this->config.customCountersEnd.size()
706  })
707  };
708 
709  // remove counter variable names that are not used
710  if(this->config.customCounters.size() > completeCounters) {
711  // remove counter variables of incomplete counters
712  this->config.customCounters.resize(completeCounters);
713 
714  incompleteCounters = true;
715  }
716 
717  // remove starting values that are not used
718  if(this->config.customCountersStart.size() > completeCounters) {
719  this->config.customCountersStart.resize(completeCounters);
720 
721  incompleteCounters = true;
722  }
723 
724  // remove ending values that are not used
725  if(this->config.customCountersEnd.size() > completeCounters) {
726  this->config.customCountersEnd.resize(completeCounters);
727 
728  incompleteCounters = true;
729  }
730 
731  // warn about incomplete counters
732  if(incompleteCounters) {
733  this->warning(
734  "\'custom.counters\', \'.start\', \'.end\' and \'.step\'"
735  " should have the same number of elements."
736  );
737 
738  this->warning("Incomplete counter(s) removed from configuration.");
739 
740  incompleteCounters = false;
741  }
742 
743  // remove step values that are not used, add one as step value where none is specified
744  if(this->config.customCountersStep.size() > completeCounters) {
745  incompleteCounters = true;
746  }
747 
748  this->config.customCountersStep.resize(completeCounters, 1);
749 
750  // remove aliases that are not used, add empty aliases where none exist
751  if(this->config.customCountersAlias.size() > completeCounters) {
752  incompleteCounters = true;
753  }
754 
755  this->config.customCountersAlias.resize(completeCounters);
756 
757  // remove alias summands that are not used, add zero as summand where none is specified
758  if(this->config.customCountersAliasAdd.size() > completeCounters) {
759  incompleteCounters = true;
760  }
761 
762  this->config.customCountersAliasAdd.resize(completeCounters, 0);
763 
764  // warn about unused properties
765  if(incompleteCounters) {
766  this->warning("Unused counter properties removed from configuration.");
767  }
768 
769  // check validity of counters
770  // (infinite counters are invalid, therefore the need to check for counter termination)
771  for(std::size_t n{1}; n <= this->config.customCounters.size(); ++n) {
772  const auto index{n - 1};
773 
774  if(
775  (
776  this->config.customCountersStep.at(index) <= 0
777  && this->config.customCountersStart.at(index)
778  < this->config.customCountersEnd.at(index)
779  )
780  ||
781  (
782  this->config.customCountersStep.at(index) >= 0
783  && this->config.customCountersStart.at(index)
784  > this->config.customCountersEnd.at(index)
785  )
786  ) {
787  const std::string counterName(this->config.customCounters.at(index));
788 
789  // delete the invalid counter
790  this->config.customCounters.erase(this->config.customCounters.begin() + index);
791  this->config.customCountersStart.erase(this->config.customCountersStart.begin() + index);
792  this->config.customCountersEnd.erase(this->config.customCountersEnd.begin() + index);
793  this->config.customCountersStep.erase(this->config.customCountersStep.begin() + index);
794  this->config.customCountersAlias.erase(this->config.customCountersAlias.begin() + index);
795  this->config.customCountersAliasAdd.erase(this->config.customCountersAliasAdd.begin() + index);
796 
797  --n;
798 
799  this->warning(
800  "Loop of counter \'"
801  + counterName
802  + "\' would be infinite, counter removed."
803  );
804  }
805  }
806 
807  // check properties of tokens
808  bool incompleteTokens{false};
809 
810  const auto completeTokens{
811  std::min({ // number of complete tokens (= min. size of arrays)
812  this->config.customTokens.size(),
813  this->config.customTokensSource.size(),
814  this->config.customTokensQuery.size()
815  })
816  };
817 
818  // remove token variable names that are not used
819  if(this->config.customTokens.size() > completeTokens) {
820  this->config.customTokens.resize(completeTokens);
821 
822  incompleteTokens = true;
823  }
824 
825  // remove token sources that are not used
826  if(this->config.customTokensSource.size() > completeTokens) {
827  this->config.customTokensSource.resize(completeTokens);
828 
829  incompleteTokens = true;
830  }
831 
832  // remove token queries that are not used
833  if(this->config.customTokensQuery.size() > completeTokens) {
834  this->config.customTokensQuery.resize(completeTokens);
835 
836  incompleteTokens = true;
837  }
838 
839  // warn about incomplete counters
840  if(incompleteTokens) {
841  this->warning(
842  "\'custom.tokens\', \'.tokens.source\' and \'.tokens.query\'"
843  " should have the same number of elements."
844  );
845 
846  this->warning("Incomplete token(s) removed from configuration.");
847 
848  incompleteTokens = false;
849  }
850 
851  // remove cookie headers that are not used, set to empty string where none is specified
852  if(this->config.customTokensCookies.size() > completeTokens) {
853  incompleteTokens = true;
854  }
855 
856  this->config.customTokensCookies.resize(completeTokens);
857 
858  // remove token expiration times that are not used, set to '0' where none is specified
859  if(this->config.customTokensKeep.size() > completeTokens) {
860  incompleteTokens = true;
861  }
862 
863  this->config.customTokensKeep.resize(completeTokens, 0);
864 
865  // remove token POST options that are not used, set to 'false' where none is specified
866  if(this->config.customTokensUsePost.size() > completeTokens) {
867  incompleteTokens = true;
868  }
869 
870  this->config.customTokensUsePost.resize(completeTokens, false);
871 
872  // warn about unused property
873  if(incompleteTokens) {
874  this->warning("Unused token properties removed from configuration.");
875  }
876 
877  // check properties of variables for dynamic redirect
878  bool incompleteVars{false};
879 
880  const auto completeVars{
881  std::min({ // number of complete variables (= min. size of all arrays)
882  this->config.redirectVarNames.size(),
883  this->config.redirectVarQueries.size(),
884  this->config.redirectVarSources.size()
885  })
886  };
887 
888  // remove redirect variable names that are not used
889  if(this->config.redirectVarNames.size() > completeVars) {
890  this->config.redirectVarNames.resize(completeVars);
891 
892  incompleteVars = true;
893  }
894 
895  // remove redirect queries that are not used
896  if(this->config.redirectVarQueries.size() > completeVars) {
897  this->config.redirectVarQueries.resize(completeVars);
898 
899  incompleteVars = true;
900  }
901 
902  // remove redirect sources that are not used
903  if(this->config.redirectVarSources.size() > completeVars) {
904  this->config.redirectVarSources.resize(completeVars);
905 
906  incompleteVars = true;
907  }
908 
909  // warn about incomplete counters
910  if(incompleteVars) {
911  this->warning(
912  "\'redirect.var.names\', \'.var.sources\' and \'.var.queries\'"
913  " should have the same number of elements."
914  );
915
