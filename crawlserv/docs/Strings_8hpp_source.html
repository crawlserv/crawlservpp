247  ) {

830  inline void sortAndRemoveDuplicates(
831  std::vector<std::string>& vectorOfStrings,
832  bool caseSensitive
833  ) {
834  if(caseSensitive) {
835  // case-sensitive sort
836  std::sort(vectorOfStrings.begin(), vectorOfStrings.end());
837 
838  // case-sensitive removal of co-occuring duplicates
839  vectorOfStrings.erase(
840  std::unique(
841  vectorOfStrings.begin(),
842  vectorOfStrings.end()
843  ),
844  vectorOfStrings.end()
845  );
846  }
847  else {
848  // case-insensitive sort
849  std::sort(
850  vectorOfStrings.begin(),
851  vectorOfStrings.end(),
852  [](const auto& s1, const auto& s2) {
853  const auto result{
854  std::mismatch(
855  s1.cbegin(),
856  s1.cend(),
857  s2.cbegin(),
858  s2.cend(),
859  [](const auto& s1, const auto& s2) {
860  return (s1 == s2) || std::tolower(s1) == std::tolower(s2);
861  }
862  )
863  };
864 
865  return result.second != s2.cend()
866  && (
867  result.first == s1.cend()
868  || std::tolower(*result.first) < std::tolower(*result.second)
869  );
870  }
871  );
872 
873  // case-insensitive removal of co-occuring duplicates
874  vectorOfStrings.erase(std::unique(vectorOfStrings.begin(), vectorOfStrings.end(),
875  [](const auto& s1, const auto& s2) {
876  return (s1.size() == s2.size()) && std::equal(s1.cbegin(), s1.cend(), s2.cbegin(),
877  [](const auto& c1, const auto& c2) {
878  return (c1 == c2) || std::tolower(c1) == std::tolower(c2);
879  }
880  );
881  }
882  ), vectorOfStrings.end());
883  }
884  }

964  inline void utfTidy(std::string& stringToTidy) {
965  // replace Unicode white spaces with spaces
966  for(const auto whitespace : utfWhitespaces) {
967  replaceAll(stringToTidy, whitespace, " ");
968  }
969 
970  // replace special ASCII characters with spaces
971  std::replace(stringToTidy.begin(), stringToTidy.end(), '\t', ' '); // horizontal tab
972  std::replace(stringToTidy.begin(), stringToTidy.end(), '\n', ' '); // line feed
973  std::replace(stringToTidy.begin(), stringToTidy.end(), '\v', ' '); // vertical tab
974  std::replace(stringToTidy.begin(), stringToTidy.end(), '\f', ' '); // form feed
975  std::replace(stringToTidy.begin(), stringToTidy.end(), '\r', ' '); // carriage return
976 
977  // replace double spaces
978  while(stringToTidy.find(" ") != std::string::npos) {
979  replaceAll(stringToTidy, " ", " ");
980  }
981 
982  // replace unnecessary spaces around punctuation
983  replaceAll(stringToTidy, " .", ".");
984  replaceAll(stringToTidy, " ,", ",");
985  replaceAll(stringToTidy, " :", ":");
986  replaceAll(stringToTidy, " ;", ";");
987  replaceAll(stringToTidy, "( ", "(");
988  replaceAll(stringToTidy, " )", ")");
989 
990  // trim result
991  trim(stringToTidy);
992  }
