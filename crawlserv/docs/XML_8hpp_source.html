284  inline void XML::parse(
285  std::string_view content,
286  bool repairCData,
287  bool repairComments,
288  std::queue<std::string>& warningsTo
289  ) {
290  // remove whitespaces at the beginning
291  std::size_t begin{0};
292 
293  while(content.size() > begin && std::isspace(content.at(begin)) != 0) {
294  ++begin;
295  }
296 
297  std::string xml(content, begin, content.length() - begin);
298 
299  // if necessary, try to tidy HTML and convert it to XML
300  if(
301  xml.size() < xmlBegin.length()
302  || xml.substr(0, xmlBegin.length()) != xmlBegin
303  ) {
304  HTML tidy;
305 
306  try {
307  tidy.tidyAndConvert(xml, this->warnings, this->errors, warningsTo);
308  }
309  catch(const HTML::Exception& e) {
310  throw XML::Exception(
311  "tidy-html5 error: "
312  + std::string(e.view())
313  );
314  }
315  }
316 
317  // try to repair CData
318  if(repairCData) {
319  cDataRepair(xml);
320  }
321 
322  // replace invalid comments
323  if(repairComments) {
324  replaceInvalidConditionalComments(xml);
325  replaceInvalidComments(xml);
326  }
327 
328  // create XML document
329  this->doc = std::make_unique<pugi::xml_document>();
330 
331  // parse XHTML with pugixml
332  std::istringstream in(xml);
333 
334  const auto result(
335  this->doc->load(in, pugi::parse_full)
336  );
337 
338  if(!result) {
339  // parsing error
340  std::string errorString{"XML parsing error: "};
341 
342  errorString += result.description();
343  errorString += " at #";
344  errorString += std::to_string(result.offset);
345  errorString += " (";
346 
347  if(result.offset > 0) {
348  errorString += "\'[...]";
349 
350  if(result.offset > numDebugCharacters) {
351  errorString += xml.substr(
352  result.offset - numDebugCharacters,
353  numDebugCharacters
354  );
355  }
356  else {
357  errorString += xml.substr(0, result.offset);
358  }
359 
360  errorString += "[!!!]";
361 
362  if(xml.size() > static_cast<std::size_t>(result.offset + numDebugCharacters)) {
363  errorString += "\'[...]";
364  errorString += xml.substr(result.offset, numDebugCharacters);
365  errorString += "[...]";
366  }
367  else {
368  errorString += "\'[...]";
369  errorString += xml.substr(result.offset);
370  }
371 
372  errorString += "\').";
373  }
374 
375  throw XML::Exception(errorString);
376  }
377  }

428  // internal static helper function: try to fix CData error (invalid ']]>' inside CData tag)
429  inline void XML::cDataRepair(std::string& content) {
430  auto pos{content.find(cDataBegin)};
431 
432  if(pos == std::string::npos) {
433  return;
434  }
435 
436  pos += cDataBegin.length();
437 
438  while(pos < content.size()) {
439  const auto next{content.find(cDataBegin, pos)};
440 
441  if(next == std::string::npos) {
442  break;
443  }
444 
445  auto last{content.rfind(cDataEnd, next - cDataEnd.length())};
446 
447  if(last != std::string::npos && last > pos) {
448  while(true) {
449  pos = content.find(cDataEnd, pos);
450 
451  if(pos < last) {
452  content.insert(pos + cDataEnd.length() - 1, 1, ' ');
453 
454  pos += cDataEnd.length() + 1;
455  }
456  else {
457  break;
458  }
459  }
460  }
461 
462  pos = next + cDataBegin.length();
463  }
464  }

466  // internal static helper function: replace invalid conditional comments (e.g. created by MS Excel)
467  inline void XML::replaceInvalidConditionalComments(std::string& content) {
468  std::size_t pos{0};
469 
470  while(pos < content.length()) {
471  // find next invalid conditional comment
472  pos = content.find(conditionalBegin, pos);
473 
474  if(pos == std::string::npos) {
475  break;
476  }
477 
478  // find end of invalid conditional comment
479  const auto end{
480  content.find(
481  conditionalEnd,
482  pos + conditionalBegin.length()
483  )
484  };
485 
486  if(end == std::string::npos) {
487  break;
488  }
489 
490  // insert commenting to make conditional comment valid (X)HTML
491  content.insert(pos + conditionalInsertOffsetBegin, conditionalInsert);
492  content.insert(
493  // (consider that "--" has already been added)
494  end + conditionalInsertOffsetEnd + conditionalInsert.length(),
495  conditionalInsert
496  );
497 
498  // replace "--" inside new comment with "=="
499  auto subPos{pos + conditionalBegin.length() + conditionalInsert.length()};
500 
501  while(subPos < end) {
502  subPos = content.find(commentCharsToReplace, subPos);
503 
504  if(subPos > end) {
505  break;
506  }
507 
508  content.replace(subPos, commentCharsToReplace.length(), commentCharsReplaceBy);
509 
510  subPos += commentCharsReplaceBy.length();
511  }
512 
513  // jump to the end of the changed conditional comment
514  pos = end + conditionalEnd.length() + 2 * conditionalInsert.length();
515  }
516 
517  // replace remaining invalid end tags
518  pos = 0;
519 
520  while(pos < content.length()) {
521  pos = content.find(conditionalEnd, pos);
522 
523  if(pos == std::string::npos) {
524  break;
525  }
526 
527  content.insert(pos + conditionalInsertOffsetStrayEnd, conditionalInsert);
528  content.insert(
529  pos + conditionalInsertOffsetEnd + conditionalInsert.length(),
530  conditionalInsert
531  );
532 
533  pos += conditionalEnd.length() + 2 * conditionalInsert.length();
534  }
535  }

538  inline void XML::replaceInvalidComments(std::string& content) {
539  std::size_t pos{0};
540 
541  while(pos < content.length()) {
542  // find next invalid comment
543  pos = content.find(invalidBegin, pos);
544 
545  if(pos == std::string::npos) {
546  break;
547  }
548 
549  // find end of invalid comment
550  const auto end{content.find(invalidEnd, pos + invalidBegin.length())};
551 
552  if(end == std::string::npos) {
553  break;
554  }
555 
556  // insert commenting to make comment valid (X)HTML
557  content.insert(pos + invalidInsertOffsetBegin, invalidInsertBegin);
558  content.insert(
559  // consider that "!--" has already been added
560  end + invalidInsertOffsetBegin + invalidInsertBegin.length(),
561  invalidInsertEnd
562  );
563 
564  // replace "--" inside new comment with "=="
565  auto subPos{pos + invalidBegin.length() + invalidInsertBegin.length()};
566 
567  while(subPos < end) {
568  subPos = content.find(commentCharsToReplace, subPos);
569 
570  if(subPos > end) {
571  break;
572  }
573 
574  content.replace(
575  subPos,
576  commentCharsToReplace.length(),
577  commentCharsReplaceBy
578  );
579

581  }
582 
583  // jump to the end of the changed comment
584  pos = end
