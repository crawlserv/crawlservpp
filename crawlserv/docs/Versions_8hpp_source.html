113  inline std::vector<StringString> getLibraryVersions() {
114  std::vector<StringString> result;
115 
116  // Boost
117  result.emplace_back(
118  "Boost",
119  std::to_string(BOOST_VERSION / boostMajor)
120  + '.'
121  + std::to_string(BOOST_VERSION / boostPatch % boostMinor)
122  + '.'
123  + std::to_string(BOOST_VERSION % boostPatch)
124  );
125 
126  // cURL
127  result.emplace_back("cURL", curl_version_info(CURLVERSION_NOW)->version);
128 
129  // GNU Aspell
130  result.emplace_back("GNU Aspell", aspell_version_string());
131 
132  // date.h (no version information available)
133  result.emplace_back("Howard E. Hinnant's date.h", "");
134 
135  // jsoncons
136  result.emplace_back(
137  "jsoncons",
138  std::to_string(JSONCONS_VERSION_MAJOR)
139  + '.'
140  + std::to_string(JSONCONS_VERSION_MINOR)
141  + '.'
142  + std::to_string(JSONCONS_VERSION_PATCH)
143  );
144 
145  // mongoose
146  result.emplace_back("mongoose", MG_VERSION);
147 
148  // MySQL Connector/C++
149  sql::Driver * driver{get_driver_instance()};
150 
151  if(driver != nullptr) {
152  result.emplace_back(
153  driver->getName(),
154  std::to_string(driver->getMajorVersion())
155  + '.'
156  + std::to_string(driver->getMinorVersion())
157  + '.'
158  + std::to_string(driver->getPatchVersion())
159  );
160  }
161 
162  // PCRE2
163  result.emplace_back(
164  "PCRE2",
165  std::to_string(PCRE2_MAJOR)
166  + '.'
167  + std::to_string(PCRE2_MINOR)
168  );
169 
170  // pugixml
171  result.emplace_back(
172  "pugixml",
173  std::to_string(PUGIXML_VERSION / pugixmlMajor)
174  + '.'
175  + std::to_string(PUGIXML_VERSION % pugixmlMajor / pugixmlMinor)
176  + '.'
177  + std::to_string(PUGIXML_VERSION % pugixmlMinor)
178  );
179 
180  // RapidJSON
181  result.emplace_back("RapidJSON", RAPIDJSON_VERSION_STRING);
182 
183  // rawr-gen (no version information available)
184  result.emplace_back("rawr-gen by Kelly Rauchenberger", "");
185 
186  // tidy-html5
187  result.emplace_back("tidy-html5", tidyLibraryVersion());
188 
189  // uriparser
190  result.emplace_back(
191  "uriparser",
192  std::to_string(URI_VER_MAJOR)
193  + '.'
194  + std::to_string(URI_VER_MINOR)
195  + '.'
196  + std::to_string(URI_VER_RELEASE)
197  + URI_VER_SUFFIX_ANSI
198  );
199 
200  // UTF8-CPP (WARNING: hard-coded version information not necessarily accurate)
201  result.emplace_back("UTF8-CPP", utf8CppVersion);
202 
203  // zlib
204  result.emplace_back("zlib", ZLIB_VERSION);
205 
206  return result;
207  }

218  inline std::string getLibraryVersionsStr(const std::string& indent) {
219  // get versions
220  const std::vector<StringString> versions(getLibraryVersions());
221 
222  // create resulting string
223  std::string result;
224 
225  for(const auto& version : versions) {
226  result += indent;
227  result += version.first;
228 
229  if(!(version.second.empty())) {
230  result += " v";
231  result += version.second;
232  }
233 
234  result += "\n";
235  }
236 
237  if(!versions.empty()) {
238  result.pop_back();
239  }
240 
241  return result;
242  }
