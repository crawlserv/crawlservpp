220  inline std::string URI::getSubUri(const std::vector<std::string>& args, bool whiteList) const {
221  if(this->domain.empty()) {
222  throw URI::Exception(
223  "Parsing::URI::getSubUri():"
224  " No domain has been specified or parsed"
225  );
226  }
227 
228  if(!(this->uri.valid())) {
229  throw URI::Exception(
230  "Parsing::URI::getSubUri():"
231  " No URI has been parsed"
232  );
233  }
234 
235  Wrapper::URIQueryList queryList;
236  const UriQueryListA * queryNext{nullptr};
237  int queryCount{0};
238  std::string queries;
239 
240  // get query string
241  if(this->uri.getc()->query.first != this->uri.getc()->query.afterLast) {
242  if(
243  uriDissectQueryMallocA(
244  queryList.getPtr(),
245  &queryCount,
246  this->uri.getc()->query.first,
247  this->uri.getc()->query.afterLast
248  )
249  == URI_SUCCESS
250  ) {
251  queryNext = queryList.getc();
252 
253  while(queryNext != nullptr) {
254  if(
255  (
256  whiteList
257  && std::find(
258  args.cbegin(),
259  args.cend(),
260  queryNext->key
261  ) != args.cend()
262  ) || (
263  !whiteList
264  && std::find(
265  args.cbegin(),
266  args.cend(),
267  queryNext->key
268  ) == args.cend()
269  )
270  ) {
271  queries += queryNext->key;
272 
273  if(queryNext->value != nullptr) {
274  queries += "=" + std::string(queryNext->value);
275  }
276 
277  queries += "&";
278  }
279 
280  queryNext = queryNext->next;
281  }
282  }
283  }
284 
285  if(!queries.empty()) {
286  queries.pop_back();
287  }
288 
289  // construct URI (starting with slash if sub-URI or with domain if website is cross-domain)
290  UriPathSegmentStructA * nextSegment = this->uri.getc()->pathHead;
291 
292  std::string result;
293 
294  if(this->crossDomain) {
295  result = URI::textRangeToString(this->uri.getc()->hostText);
296 
297  if(result.empty()) {
298  return "";
299  }
300  }
301 
302  while(nextSegment != nullptr) {
303  result += "/" + URI::unescape(URI::textRangeToString(nextSegment->text), false);
304 
305  nextSegment = nextSegment->next;
306  }
307 
308  // add queries
309  if(!queries.empty()) {
310  result += "?" + queries;
311  }
312 
313  return result;
314  }

474  inline bool URI::parseLink(std::string_view uriToParse) {
475  // reset old URI if necessary
476  this->uri.clear();
477 
478  // error checking
479  if(this->domain.empty()) {
480  throw URI::Exception(
481  "Parsing::URI::parseLink():"
482  " No domain has been specified or parsed"
483  );
484  }
485 
486  if(this->subUri.empty()) {
487  throw URI::Exception(
488  "Parsing::URI::parseLink():"
489  " No sub-URI has been parsed"
490  );
491  }
492 
493  // copy URI
494  std::string linkCopy(uriToParse);
495 
496  // remove anchor if necessary
497  const auto end{linkCopy.find('#')};
498 
499  if(end != std::string::npos && linkCopy.size() > end) {
500  if(end > 0) {
501  linkCopy = linkCopy.substr(0, end);
502  }
503  else {
504  linkCopy = "";
505  }
506  }
507 
508  // trim and escape URI
509  Helper::Strings::trim(linkCopy);
510 
511  linkCopy = URI::escapeUri(linkCopy);
512 
513  // check for empty link
514  if(linkCopy.empty()) {
515  return false;
516  }
517 
518  // create new URI
519  this->uri.create();
520 
521  // create temporary URI for relative link
522  Wrapper::URI relativeSource;
523 
524  relativeSource.create();
525 
526  // NOTE: URI needs to be stored in-class BEFORE PARSING to provide long-term access by the parsing library
527  // (otherwise the URI would be out of scope for the library after leaving this member function)
528  this->link.swap(linkCopy);
529 
530  // parse relative link
531  const char * errorPos{nullptr};
532  const auto errorCode{
533  uriParseSingleUriA(
534  relativeSource.get(),
535  this->link.c_str(),
536  &errorPos
537  )
538  };
539 
540  if(errorCode != URI_SUCCESS) {
541  const std::string rest(errorPos);
542 
543  std::string errorString{
544  "Parsing::URI::parseLink():"
545  " URI Parser error #"
546  };
547 
548  errorString += std::to_string(errorCode);
549  errorString += ": \'";
550 
551  if(rest.size() < this->link.size()) {
552  errorString += this->link.substr(0, this->link.size() - rest.size());
553  errorString += "[!!!]";
554  errorString += rest;
555  }
556  else {
557  errorString += this->link;
558  errorString += "[!!!]";
559  }
560 
561  errorString += "\'";
562 
563  throw URI::Exception(errorString);
564  }
565 
566  // resolve reference
567  if(
568  uriAddBaseUriExA(
569  this->uri.get(),
570  relativeSource.getc(),
571  this->base.getc(),
572  URI_RESOLVE_IDENTICAL_SCHEME_COMPAT
573  ) != URI_SUCCESS
574  ) {
575  throw URI::Exception(
576  "Parsing::URI::parseLink():"
577  " Reference resolving failed for \'"
578  + URI::toString(relativeSource)
579  + "\'"
580  );
581  }
582 
583  // normalize URI
584  const auto dirtyParts{
585  uriNormalizeSyntaxMaskRequiredA(
586  this->uri.getc()
587  )
588  };
589 
590  if(
591  dirtyParts != URI_NORMALIZED
592  && uriNormalizeSyntaxExA(
593  this->uri.get(),
594  dirtyParts
595  ) != URI_SUCCESS
596  ) {
597  throw URI::Exception(
598  "Parsing::URI::parseLink():"
599  " Normalizing failed for \'"
600  + URI::toString(this->uri)
601  + "\'"
602  );
603  }
604 
605  return true;
606  }

684  inline std::string URI::escapeUri(std::string_view uriToEscape) {
685  std::string result;
686  size_t pos{0};
687 
688  while(pos < uriToEscape.size()) {
689  auto end{uriToEscape.find_first_of(";/?:@=&#%", pos)};
690 
691  if(end == std::string::npos) {
692  end = uriToEscape.size();
693  }
694 
695  if(end - pos > 0) {
696  const std::string part(uriToEscape, pos, end - pos);
697 
698  result += URI::escape(part, false);
699  }
700 
701  if(end < uriToEscape.size()) {
702  result += uriToEscape.at(end);
703  }
704 
705  pos = end + 1;
706  }
707 
708  // replace % with %25 if not followed by a two-digit hexadecimal number
709  Helper::Strings::encodePercentage(result);
710 
711  return result;
712  }

729  inline void URI::makeAbsolute(std::string_view uriBase, std::vector<std::string>& uris) {
730  // create base URI
731  Wrapper::URI baseUri;
732 
733  baseUri.create();
734 
735  // parse base URI
736  const char * errorPos{nullptr};
737  const auto errorCode{
738  uriParseSingleUriExA(
739  baseUri.get(),
740  uriBase.data(),
741  uriBase.data() + uriBase.size(),
742  &errorPos
743  )
744  };
745 
746  if(errorCode != URI_SUCCESS) {
747  const std::string end(
748  errorPos,
749  uriBase.size() - (errorPos - uriBase.data())
750  );
751 
752  std::string errorString{
753  "Parsing::URI::makeAbsolute():"
754  " URI Parser error #"
755  };
756 
757  errorString += std::to_string(errorCode);
758  errorString += ": \'";
759 
760  if(end.size() < uriBase.size()) {
761  errorString += uriBase.substr(0, uriBase.size() - end.size());
762  errorString += "[!!!]";
763  errorString += end;
764  }
765  else {
766  errorString += uriBase;
767  errorString += "[!!!]";
768  }
769 
770  errorString += "\'";
771 
772  throw URI::Exception(errorString);
773  }
774 
775  // go through (possibly) relative URIs
776  std::vector<std::string> result;
777 
778  for(const auto& relUri : uris) {
779  // skip empty URIs
780  if(relUri.empty()) {
781  continue;
782  }
783 
784  // create relative URI
785  Wrapper::URI relativeSource;
786 
787  relativeSource.create();
788 
789  // create absolute URI
790  Wrapper::URI absoluteDest;
791 
792  absoluteDest.create();
793 
794  // parse relative URI
795  if(uriParseSingleUriA(
796  relativeSource.get(),
797  relUri.c_str(),
798  nullptr
799  ) != URI_SUCCESS) {
800  // ignore single URIs that cannot be parsed
801  continue;
802  }
803 
804  // resolve reference
805  if(uriAddBaseUriExA(
806  absoluteDest.get(),
807  relativeSource.getc(),
808  baseUri.getc(),
809  URI_RESOLVE_IDENTICAL_SCHEME_COMPAT
810  ) != URI_SUCCESS) {
811  // ignore single URIs that cannot be resolved
812  continue;
813  }
814 
815  // normalize absolute URI
816  const auto dirtyParts{
817  uriNormalizeSyntaxMaskRequiredA(
818  absoluteDest.getc()
819  )
820  };
821 
822  if(
823  dirtyParts != URI_NORMALIZED
824  && uriNormalizeSyntaxExA(
825  absoluteDest.get(),
826  dirtyParts
827  ) != URI_SUCCESS
828  ) {
829  // ignore single URIs that cannot be normalized
830  continue;
831  }
832 
833  // add normalized absolute URI
834  result.emplace_back(URI::toString(absoluteDest));
835  }
836 
837  // swap (possibly) relative with absolute URIs
838  result.swap(uris);
839  }
