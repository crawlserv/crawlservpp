256  inline void convertCustomDateTimeToSQLTimeStamp(
257  std::string& dateTime,
258  const std::string& customFormat
259  ) {
260  // check arguments
261  if(dateTime.empty()) {
262  return;
263  }
264 
265  if(customFormat.empty()) {
266  throw Exception(
267  "DateTime::convertCustomDateTimeToSQLTimeStamp():"
268  " No custom format specified"
269  );
270  }
271 
272  // check for UNIX time format
273  if(
274  customFormat == unixTimeFormat
275  || (
276  customFormat.length() > unixTimeFormatXLength
277  && (
278  customFormat.substr(0, unixTimeFormatXLength) == unixTimeFormatPlus
279  || customFormat.substr(0, unixTimeFormatXLength) == unixTimeFormatMinus
280  )
281  )
282  ) {
283  // get (optional) offset from UNIX time
284  std::int64_t offset{0};
285 
286  if(customFormat.length() > unixTimeFormatXLength) {
287  try {
288  offset = std::stol(customFormat.substr(unixTimeFormatXOffset));
289  }
290  catch(const std::exception& e) {
291  throw Exception(
292  "DateTime::convertCustomDateTimeToSQLTimeStamp(): Invalid date/time format - "
293  + customFormat
294  + " [expected: UNIX, UNIX+N or UNIX-N where N is a valid number]"
295  );
296  }
297  }
298 
299  // get UNIX time
300  std::time_t unixTime{0};
301 
302  try {
303  if(dateTime.find('.') != std::string::npos) {
304  // handle values with comma as floats (and round them)
305  float f{std::stof(dateTime)};
306 
307  unixTime = static_cast<std::time_t>(std::lround(f));
308  }
309  else {
310  unixTime = static_cast<std::time_t>(std::stol(dateTime));
311  }
312  }
313  catch(const std::exception& e) {
314  throw Exception(
315  "Could not convert \'"
316  + dateTime
317  + "\' [expected format: \'"
318  + customFormat
319  + "\'] to date/time "
320  );
321  }
322 
323  // remove (optional) offset
324  unixTime -= offset;
325 
326  // conversion
327  dateTime = date::format(
328  sqlTimeStamp,
329  std::chrono::system_clock::from_time_t(unixTime)
330  );
331  }
332  else {
333  // remove English ordinal endings (st, nd, rd, th)
334  removeEnglishOrdinals(dateTime);
335 
336  std::istringstream in(dateTime);
337  date::sys_seconds tp;
338 
339  in >> date::parse(customFormat, tp);
340 
341  if(bool(in)) {
342  dateTime = date::format(sqlTimeStamp, tp);
343  }
344  else {
345  // try good old C time
346  struct std::tm cTime{};
347 
348  {
349  std::istringstream inStringStream(dateTime);
350 
351  inStringStream.imbue(std::locale(std::setlocale(LC_ALL, nullptr)));
352 
353  inStringStream >> std::get_time(&cTime, customFormat.c_str());
354 
355  if(inStringStream.fail()) {
356  throw Exception(
357  "Could not convert '"
358  + dateTime
359  + "' [expected format: '"
360  + customFormat
361  + "'] to date/time "
362  );
363  }
364  }
365 
366  std::array<char, sqlTimeStampLength + 1> out{};
367 
368  const auto len{
369  std::strftime(out.data(), sqlTimeStampLength + 1, sqlTimeStamp, &cTime)
370  };
371 
372  if(len > 0) {
373  dateTime = std::string(out.data(), len);
374  }
375  else {
376  throw Exception(
377  "Could not convert '"
378  + dateTime
379  + "' [expected format: '"
380  + customFormat
381  + "'] to date/time "
382  );
383  }
384  }
385  }
386  }

559  inline std::string microsecondsToString(std::uint64_t microseconds) {
560  auto rest{microseconds};
561  std::string result;
562 
563  const auto days{
564  rest / microsecondsPerDay
565  };
566 
567  rest -= days * microsecondsPerDay;
568 
569  // narrowing cast as hours are never more than hours per day
570  const auto hours{static_cast<std::uint8_t>(rest / microsecondsPerHour)};
571 
572  rest -= hours * microsecondsPerHour;
573 
574  // narrowing cast as minutes are never more than minutes per hour
575  const auto minutes{static_cast<std::uint8_t>(rest / microsecondsPerMinute)};
576 
577  rest -= minutes * microsecondsPerMinute;
578 
579  // narrowing cast as seconds are never more than seconds per minute
580  const auto seconds{static_cast<std::uint8_t>(rest / microsecondsPerSecond)};
581 
582  rest -= seconds * microsecondsPerSecond;
583 
584  // narrowing cast as milliseconds are never more than milliseconds per second
585  const auto milliseconds{static_cast<std::uint16_t>(rest / microsecondsPerMillisecond)};
586 
587  rest -= milliseconds * microsecondsPerMillisecond;
588 
589  if(days > 0) {
590  result += std::to_string(days) + "d ";
591  }
592 
593  if(hours > 0) {
594  result += std::to_string(hours) + "h ";
595  }
596 
597  if(minutes > 0) {
598  result += std::to_string(minutes) + "min ";
599  }
600 
601  if(seconds > 0) {
602  result += std::to_string(seconds) + "s ";
603  }
604 
605  if(milliseconds > 0) {
606  result += std::to_string(milliseconds) + "ms ";
607  }
608 
609  if(rest > 0) {
610  result += std::to_string(rest) + "Î¼s ";
611  }
612 
613  if(result.empty()) {
614  return "<1Î¼s";
615  }
616 
617  // remove last space
618  result.pop_back();
619 
620  return result;
621  }

789  inline bool isISODateInRange(
790  std::string_view isoDate,
791  std::string_view rangeFrom,
792  std::string_view rangeTo
793  ) {
794  if(isoDate.length() < isoDateLength) {
795  return false;
796  }
797 
798  if(rangeFrom.length() < isoDateLength && rangeTo.length() < isoDateLength) {
799  return true;
800  }
801 
802  if(rangeTo.length() < isoDateLength) {
803  return isoDate.substr(0, isoDateLength) >= rangeFrom.substr(0, isoDateLength);
804  }
805 
806  if(rangeFrom.length() < isoDateLength) {
807  return isoDate.substr(0, isoDateLength) <= rangeTo.substr(0, isoDateLength);
808  }
809 
810  return isoDate.substr(0, isoDateLength) >= rangeFrom.substr(0, isoDateLength)
811  && isoDate <= rangeTo.substr(0, isoDateLength);
812  }

819  inline void removeEnglishOrdinals(std::string& strInOut) {
820  std::size_t pos{0};
821 
822  while(pos + englishOrdinalSuffixLength <= strInOut.length()) {
823  std::size_t next{std::string::npos};
824 
825  for(const auto& suffix : englishOrdinalSuffixes) {
826  const auto search{strInOut.find(suffix, pos)};
827 
828  if(search < next) {
829  next = search;
830  }
831  }
832 
833  pos = next;
834 
835  if(pos == std::string::npos) {
836  break;
837  }
838 
839  if(pos > 0) {
840  if(std::isdigit(strInOut.at(pos - 1)) != 0) {
841  const auto end{pos + englishOrdinalSuffixLength};
842 
843  if(
844  end == strInOut.length()
845  || std::isspace(strInOut.at(end)) != 0
846  || std::ispunct(strInOut.at(end)) != 0
847  ) {
848  // remove st, nd, rd or th
849  strInOut.erase(pos, englishOrdinalSuffixLength);
850 
851  // skip whitespace/punctuation afterwards
852  ++pos;
853  }
854  else {
855  pos += englishOrdinalSuffixLength + 1;
856  }
857  }
858  else {
859  pos += englishOrdinalSuffixLength + 1;
860  }
861  }
862  else {
863  pos += englishOrdinalSuffixLength + 1;
864  }
865  }
866  }
