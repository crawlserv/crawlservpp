127  inline std::string iso88591ToUtf8(std::string_view strIn) {
128  std::string strOut;
129 
130  // guess maximum memory requirement
131  strOut.reserve(strIn.size() * utf8MemoryFactor);
132 
133  for(const uint8_t c : strIn) {
134  if(c < bitmaskTopBit) {
135  strOut.push_back(c);
136  }
137  else {
138  strOut.push_back(bitmaskTopTwoBits | c >> shiftSixBits);
139  strOut.push_back(bitmaskTopBit | (c & bitmaskLastSixBits0b000001));
140  }
141  }
142 
143  return strOut;
144  }

188  inline bool isLastCharValidUtf8(const std::string& stringToCheck) {
189  if(stringToCheck.empty()) {
190  return true;
191  }
192 
193  // check for valid one-byte character
194  auto pos{stringToCheck.size() - 1};
195 
196  if(utf8::is_valid(stringToCheck.substr(pos, oneByte))) {
197  return true;
198  }
199 
200  if(stringToCheck.size() < twoBytes) {
201  return false;
202  }
203 
204  --pos;
205 
206  // check for valid two-byte character
207  if(utf8::is_valid(stringToCheck.substr(pos, twoBytes))) {
208  return true;
209  }
210 
211  if(stringToCheck.size() < threeBytes) {
212  return false;
213  }
214 
215  --pos;
216 
217  // check for valid three-byte character
218  if(utf8::is_valid(stringToCheck.substr(pos, threeBytes))) {
219  return true;
220  }
221 
222  if(stringToCheck.size() < fourBytes) {
223  return false;
224  }
225 
226  --pos;
227 
228  // check for valid four-byte character
229  if(utf8::is_valid(stringToCheck.substr(pos, fourBytes))) {
230  return true;
231  }
232 
233  return false;
234  }

258  inline bool repairUtf8(std::string_view strIn, std::string& strOut) {
