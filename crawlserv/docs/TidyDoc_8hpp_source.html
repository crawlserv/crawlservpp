195  inline std::string TidyDoc::getOutput(std::queue<std::string>& warningsTo) {
196  TidyBuffer buffer;
197 
198  switch(tidySaveBuffer(this->doc, buffer.get())) {
199  case 0:
200  // everything went fine
201  break;
202 
203  case 1:
204  case 2:
205  // warnings or errors occured
206  if(this->errors.valid()) {
207  std::queue<std::string> warnings(
208  Helper::Strings::splitToQueue(
209  this->errors.getString(),
210  '\n',
211  true
212  )
213  );
214 
215  while(!warnings.empty()) {
216  warningsTo.emplace(warnings.front());
217 
218  warnings.pop();
219  }
220 
221  this->errors.clear();
222  }
223 
224  break;
225 
226  default:
227  // fatal errors occured
228  if(this->errors.valid() && !(this->errors.empty())) {
229  throw Exception("Could not write to buffer: " + errors.getString());
230  }
231 
232  throw Exception("Could not write to buffer");
233  }
234 
235  if(buffer.valid() && !buffer.empty()) {
236  return buffer.getString();
237  }
238 
239  return std::string();
240  }
