134  inline Curl::Curl() {
135  // initialize global instance of libcurl if necessary
136  if(globalInit) {
137  this->localInit = false;
138  }
139  else {
140  //NOLINTNEXTLINE(hicpp-signed-bitwise)
141  const auto curlCode{curl_global_init(CURL_GLOBAL_ALL)};
142 
143  if(curlCode != CURLE_OK) {
144  throw std::runtime_error(curl_easy_strerror(curlCode));
145  }
146 
147  globalInit = true;
148 
149  this->localInit = true;
150  }
151 
152  // initialize local instance of libcurl
153  try {
154  this->init();
155  }
156  catch(...) {
157  if(this->localInit) {
158  if(globalInit) {
159  curl_global_cleanup();
160 
161  globalInit = false;
162  }
163 
164  this->localInit = false;
165  }
166 
167  throw;
168  }
169  }
