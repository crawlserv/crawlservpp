723  inline bool Container::nextSubSet() {
724  // check subsets
725  if(this->subSetNumber < this->subSetCurrent) {
726  throw Exception(
727  "Query::Container::nextSubSet():"
728  " Invalid subset selected"
729  );
730  }
731 
732  if(this->subSetNumber == this->subSetCurrent) {
733  return false;
734  }
735 
736  // clear previous subset
737  if(this->subSetCurrent > 0) {
738  if(!(this->stringifiedSubSets.empty())) {
739  std::string().swap(
740  this->stringifiedSubSets.at(
741  this->subSetCurrent - 1
742  )
743  );
744  }
745 
746  switch(this->subSetType) {
747  case QueryStruct::typeXPath:
748  this->xPathSubSets.at(
749  this->subSetCurrent - 1
750  ).clear();
751 
752  break;
753 
754  case QueryStruct::typeJsonPointer:
755  rapidjson::Value(rapidjson::kObjectType).Swap(
756  this->jsonPointerSubSets.at(
757  this->subSetCurrent - 1
758  )
759  );
760 
761  break;
762 
763  case QueryStruct::typeJsonPath:
764  jsoncons::json().swap(
765  this->jsonPathSubSets.at(
766  this->subSetCurrent - 1
767  )
768  );
769 
770  break;
771 
772  default:
773  break;
774  }
775  }
776 
777  // increment index (+ 1) of current subset
778  ++(this->subSetCurrent);
779 
780  return true;
781  }
