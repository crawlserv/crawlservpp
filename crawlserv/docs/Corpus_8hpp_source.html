466  bool deleteInputData

469  this->clear();
470 
471  // concatenate texts
472  for(auto& text : texts) {
473  // add text to corpus
474  this->corpus += text;
475 
476  if(deleteInputData) {
477  // free memory early
478  std::string().swap(text);
479  }
480 
481  // add space at the end of the corpus
482  this->corpus.push_back(' ');
483  }
484 
485  if(deleteInputData) {
486  // free memory early
487  std::vector<std::string>().swap(texts);
488  }
489 
490  // remove last space if necessary
491  if(!(this->corpus.empty())) {
492  this->corpus.pop_back();

515  inline void Corpus::create(
516  std::vector<std::string>& texts,
517  std::vector<std::string>& articleIds,
518  std::vector<std::string>& dateTimes,
519  bool deleteInputData
520  ) {
521  // check arguments
522  if(articleIds.empty() && dateTimes.empty()) {
523  this->create(texts, deleteInputData);
524 
525  return;
526  }
527 
528  // clear old corpus
529  this->clear();
530 
531  Struct::TextMapEntry dateMapEntry;
532 
533  for(std::size_t n{0}; n < texts.size(); ++n) {
534  auto pos{this->corpus.size()};
535  auto& text{texts.at(n)};
536 
537  // add article ID (or empty article) to article map
538  if(articleIds.size() > n) {
539  auto& articleId{articleIds.at(n)};
540 
541  this->articleMap.emplace_back(pos, text.length(), articleId);
542 
543  if(deleteInputData && !articleId.empty()) {
544  // free memory early
545  std::string().swap(articleId);
546  }
547  }
548  else if(!(this->articleMap.empty()) && this->articleMap.back().value.empty()) {
549  // expand empty article in the end of the article map
550  this->articleMap.back().length += text.length() + 1; /* include space before current text */
551  }
552  else {
553  // add empty article to the end of the article map
554  this->articleMap.emplace_back(pos, text.length());
555  }
556 
557  // add date to date map if necessary
558  if(dateTimes.size() > n) {
559  auto& dateTime{dateTimes.at(n)};
560 
561  // check for valid (long enough) date/time
562  if(dateTime.length() >= dateLength) {
563  // get only date (YYYY-MM-DD) from date/time
564  const std::string date(dateTime, 0, dateLength);
565 
566  // check whether a date is already set
567  if(!dateMapEntry.value.empty()) {
568  // date is already set -> compare with current date
569  if(dateMapEntry.value == date) {
570  // last date equals current date -> append text to last date
571  dateMapEntry.length += text.length() + 1; /* include space before current text */
572  }
573  else {
574  // last date differs from current date -> conclude last date and start new date
575  this->dateMap.emplace_back(dateMapEntry);
576 
577  dateMapEntry = Struct::TextMapEntry(this->corpus.size(), text.length(), date);
578  }
579  }
580  else {
581  // no date is set yet -> start new date
582  dateMapEntry = Struct::TextMapEntry(this->corpus.size(), text.length(), date);
583  }
584  }
585  else if(!dateMapEntry.value.empty()) {
586  // no valid date found, but last date is set -> conclude last date
587  this->dateMap.emplace_back(dateMapEntry);
588 
589  dateMapEntry = Struct::TextMapEntry();
590  }
591 
592  if(deleteInputData && !dateTime.empty()) {
593  // free memory early
594  std::string().swap(dateTime);
595  }
596  }
597 
598  // concatenate corpus text
599  this->corpus += text;
600 
601  if(deleteInputData) {
602  // free memory early
603  std::string().swap(text);
604  }
605 
606  // add space at the end of the corpus
607  this->corpus.push_back(' ');
608  }
609 
610  if(deleteInputData) {
611  // free memory early
612  std::vector<std::string>().swap(texts);
613  std::vector<std::string>().swap(articleIds);
614  std::vector<std::string>().swap(dateTimes);
615  }
616 
617  // finish up corpus data
618  if(!(this->corpus.empty())) {
619  // remove last space
620  this->corpus.pop_back();
621  }
622 
623  // check for unfinished date
624  if(!dateMapEntry.value.empty()) {
625  // conclude last date
626  this->dateMap.emplace_back(dateMapEntry);
627  }
628  }

655  inline void Corpus::combine(
656  std::vector<std::string>& chunks,
657  std::vector<Struct::TextMap>& articleMaps,
658  std::vector<Struct::TextMap>& dateMaps,
659  bool deleteInputData
660  ) {
661  // clear old corpus
662  this->clear();
663 
664  // reserve memory
665  const auto size{
666  std::accumulate(
667  chunks.cbegin(),
668  chunks.cend(),
669  static_cast<std::size_t>(0),
670  [](const auto& a, const auto& b) {
671  return a + b.size();
672  }
673  )
674  };
675 
676  this->corpus.reserve(size);
677 
678  // add chunks
679  for(std::size_t n{0}; n < chunks.size(); ++n) {
680  // save current position in new corpus
681  const auto pos{this->corpus.size()};
682 
683  // add text of chunk to corpus
684  this->corpus += chunks.at(n);
685 
686  if(deleteInputData) {
687  // free memory early
688  std::string().swap(chunks.at(n));
689  }
690 
691  bool beginsWithNewArticle{false};
692 
693  if(articleMaps.size() > n) {
694  // add article map
695  auto& map{articleMaps.at(n)};
696 
697  if(!map.empty()) {
698  const auto& first{map.at(0)};
699 
700  // consistency check
701  if(this->checkConsistency && first.pos > 1) {
702  throw Exception(
703  "Corpus::combine(): Article map in corpus chunk starts at #"
704  + std::to_string(first.pos)
705  + " instead of #0 or #1"
706  );
707  }
708 
709  auto it{map.cbegin()};
710 
711  // compare first new article ID with last one
712  if(!(this->articleMap.empty()) && this->articleMap.back().value == first.value) {
713  // append current article to last one
714  this->articleMap.back().length += first.length;
715 
716  ++it;
717  }
718  else {
719  beginsWithNewArticle = true;
720  }
721 
722  // add remaining articles to map
723  for(; it != map.cend(); ++it) {
724  this->articleMap.emplace_back(pos + it->pos, it->length, it->value);
725  }
726 
727  if(deleteInputData) {
728  // free memory early
729  Struct::TextMap().swap(map);
730  }
731  }
732  }
733 
734  if(dateMaps.size() > n) {
735  // add date map
736  auto& map{dateMaps.at(n)};
737 
738  if(!map.empty()) {
739  const auto& first{map.at(0)};
740  auto it{map.cbegin()};
741 
742  // compare first new date with last one
743  if(!(this->dateMap.empty()) && this->dateMap.back().value == first.value) {
744  // append current date to last one
745  this->dateMap.back().length += first.length;
746 
747  // add missing space between articles if chunk begins with a new article and the date has been extended
748  if(beginsWithNewArticle) {
749  ++(this->dateMap.back().length);
750  }
751 
752  ++it;
753  }
754 
755  for(; it != map.cend(); ++it) {
756  this->dateMap.emplace_back(pos + it->pos, it->length, it->value);
757  }
758 
759  if(deleteInputData) {
760  // free memory early
761  Struct::TextMap().swap(map);
762  }
763  }
764  }
765  }
766 
767  if(deleteInputData) {
768  // free memory early
769  std::vector<std::string>().swap(chunks);
770  std::vector<Struct::TextMap>().swap(articleMaps);
771  std::vector<Struct::TextMap>().swap(dateMaps);
772  }
773  }

843  // copy corpus, article map and date map in chunks of specified size and return whether slicing was successful,
844  // throws Corpus::Exception
845  inline void Corpus::copyChunks(
846  std::size_t chunkSize,
847  std::vector<std::string>& to,
848  std::vector<Struct::TextMap>& articleMapsTo,
849  std::vector<Struct::TextMap>& dateMapsTo
850  ) const {
851  // check arguments
852  if(chunkSize == 0) {
853  if(this->corpus.empty()) {
854  return;
855  }
856 
857  throw Exception(
858  "Corpus::copyChunks():"
859  " Invalid chunk size zero for non-empty corpus"
860  );
861  }
862 
863  // check whether slicing is necessary
864  if(this->corpus.size() <= chunkSize) {
865  to.emplace_back(this->corpus);
866  articleMapsTo.emplace_back(this->articleMap);
867  dateMapsTo.emplace_back(this->dateMap);
868 
869  return;
870  }
871 
872  // reserve probable memory for chunks
873  const auto chunks{
874  this->corpus.size() / chunkSize
875  + (this->corpus.size() % chunkSize > 0 ? 1 : 0)
876  };
877 
878 
879  to.reserve(chunks);
880 
881  if(!(this->articleMap.empty())) {
882  articleMapsTo.reserve(chunks);
883  }
884 
885  if(!(this->dateMap.empty())) {
886  dateMapsTo.reserve(chunks);
887  }
888 
889  // slice corpus into chunks
890  bool noSpace{false};
891 
892  if(this->articleMap.empty()) {
893  // no article part: simply add parts of the corpus
894  std::size_t pos{0};
895 
896  while(pos < this->corpus.size()) {
897  to.emplace_back(this->corpus, pos, this->getValidLengthOfSlice(pos, chunkSize, chunkSize));
898 
899  pos += to.back().size();
900  }
901 
902  noSpace = true;
903  }
904  else {
905  std::size_t corpusPos{0};
906  std::size_t articlePos{0};
907  auto articleIt{this->articleMap.cbegin()};
908  auto dateIt{this->dateMap.cbegin()};
909 
910  while(corpusPos < this->corpus.size()) { /* loop for chunks */
911  // create chunk
912  Struct::TextMap chunkArticleMap;
913  Struct::TextMap chunkDateMap;
914  std::string chunk;
915 
916  // add space if necessary
917  if(noSpace) {
918  chunk.push_back(' ');
919 
920  ++corpusPos;
921 
922  noSpace = false;
923  }
924 
925  for(; articleIt != this->articleMap.cend(); ++articleIt) { /* loop for multiple articles inside one chunk */
926  if(dateIt != this->dateMap.cend()) {
927  // check date of the article
928  if(
929  articlePos == 0
930  && articleIt->pos > dateIt->pos + dateIt->length
931  ) {
932  ++dateIt;
933  }
934 
935  if(this->checkConsistency && articleIt->pos > dateIt->pos + dateIt->length) {
936  throw Exception(
937  "Article position (#"
938  + std::to_string(articleIt->pos)
939  + ") lies behind date at [#"
940  + std::to_string(dateIt->pos)
941  + ";#"
942  + std::to_string(dateIt->pos + dateIt->length)
943  + "]"
944  );
945  }
946  }
947 
948  // get remaining article length
949  const auto remaining{articleIt->length - articlePos};
950 
951  if(chunk.size() + remaining <= chunkSize) {
952  if(remaining > 0) {
953  // add the remainder of the article to the chunk
954  chunkArticleMap.emplace_back(chunk.size(), remaining, articleIt->value);
955 
956  if(dateIt != this->dateMap.cend()) {
957  if(!chunkDateMap.empty() && chunkDateMap.back().value == dateIt->value) {
958  chunkDateMap.back().length += remaining + 1; /* including space before article */
959  }
960  else if(corpusPos >= dateIt->pos) {
961  chunkDateMap.emplace_back(chunk.size(), remaining, dateIt->value);
962  }
963  }
964 
965  chunk.append(this->corpus, corpusPos, remaining);
966 
967  // update position in corpus
968  corpusPos += remaining;
969  }
970 
971  // reset position in (next) article
972  articlePos = 0;
973 
974  if(chunk.size() < chunkSize) {
975  // add space to the end of the article
976  chunk.push_back(' ');
977 
978  ++corpusPos;
979 
980  // check for end of chunk
981  if(chunk.size() == chunkSize) {
982  // start next chunk with next article
983  ++articleIt;
984 
985  break; // chunk is full
986  }
987  }
988  else {
989  // add space to the beginning of the next chunk instead
990  noSpace = true;
991 
992  // start next chunk with next article
993  ++articleIt;
994 
995  break; // chunk is full
996  }
997  }
998  else {
999  // fill the remainder of the chunk with part of the article
1000  auto fill{chunkSize - chunk.size()};
1001 
1002  if(fill == 0) {
1003  break; /* chunk is full */
1004  }
1005 
1006  // check the end for valid UTF-8
1007  fill = this->getValidLengthOfSlice(corpusPos, fill, chunkSize);
1008 
1009  if(fill == 0) {
1010  break; /* not enough space in chunk for last (UTF-8) character */
1011  }
1012 
1013  chunkArticleMap.emplace_back(chunk.size(), fill, articleIt->value);
1014 
1015  if(dateIt != this->dateMap.cend()) {
1016  if(!chunkDateMap.empty() && chunkDateMap.back().value == dateIt->value) {
1017  chunkDateMap.back().length += fill + 1; /* including space before the article */
1018  }
1019  else if(corpusPos >= dateIt->pos) {
1020  chunkDateMap.emplace_back(chunk.size(), fill, dateIt->value);
1021  }
1022  }
1023 
1024  chunk.append(this->corpus, corpusPos, fill);
1025 
1026  // update positions
1027  corpusPos += fill;
1028  articlePos += fill;
1029 
1030  break; /* chunk is full */
1031  }
1032  }
1033 
1034  // consistency checks
1035  if(this->checkConsistency) {
1036  if(chunk.size() > chunkSize) {
1037  throw Exception(
1038  "Corpus::copyChunks(): Chunk is too large: "
1039  + std::to_string(chunk.size())
1040  + " > "
1041  + std::to_string(chunkSize)
1042  );
1043  }
1044 
1045  if(articleIt == this->articleMap.cend() && corpusPos < this->corpus.size()) {
1046  throw Exception(
1047  "Corpus::copyChunks(): End of articles, but not of corpus: #"
1048  + std::to_string(corpusPos)
1049  + " < #"
1050  + std::to_string(this->corpus.size())
1051  );
1052  }
1053  }
1054 
1055  // check for empty chunk (should not happen)
1056  if(chunk.empty()) {
1057  break;
1058  }
1059 
1060  // add current chunk
1061  to.emplace_back(chunk);
1062  articleMapsTo.emplace_back(chunkArticleMap);
1063  dateMapsTo.emplace_back(chunkDateMap);
1064  }
1065  }
1066 
1067  if(!(this->articleMap.empty()) && !to.empty()) {
1068  // consistency check
1069  if(this->checkConsistency && to.back().empty()) {
1070  throw Exception("Corpus::copyChunks(): End chunk is empty");
1071  }
1072 
1073  // remove last space
1074  if(!noSpace) {
1075  to.back().pop_back();
1076  }
