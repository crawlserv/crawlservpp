127  inline JsonPointer::JsonPointer(const std::string& pointerString, bool textOnlyQuery) : textOnly(textOnlyQuery) {
128  // copy and trim pointer string
129  std::string string{pointerString};
130 
131  Helper::Strings::trim(string);
132 
133  if(string.empty()) {
134  throw Exception("No JSONPointer string given");
135  }
136 
137  // check whether multiple JSONPointers need to be constructed
138  if(string.find("$$") != std::string::npos) {
139  this->pointerStringMulti = string;
140  }
141 
142  if(this->pointerStringMulti.empty()) {
143  this->pointerFirst = rapidjson::Pointer(string);
144 
145  if(!(this->pointerFirst.IsValid())) {
146  throw JsonPointer::Exception("Invalid JSONPointer \'" + string + "\'");
147  }
148  }
149  else {
150  Helper::Strings::replaceAll(string, "$$", "0");
151 
152  this->pointerFirst = rapidjson::Pointer(string);
153 
154  if(!(this->pointerFirst.IsValid())) {
155  throw JsonPointer::Exception("Invalid JSONPointer \'" + string + "\'");
156  }
157  }
158  }

174  inline bool JsonPointer::getBool(const rapidjson::Document& doc) const {
175  // check document and pointer
176  if(doc.HasParseError()) {
177  throw JsonPointer::Exception("JSON parsing error");
178  }
179 
180  if(!(this->pointerFirst.IsValid())) {
181  throw JsonPointer::Exception("Invalid JSONPointer");
182  }
183 
184  // evaluate query with boolean result
185  return this->pointerFirst.Get(doc) != nullptr;
186  }

211  inline void JsonPointer::getFirst(const rapidjson::Document& doc, std::string& resultTo) const {
212  // empty result
213  resultTo.clear();
214 
215  // check document and pointer
216  if(doc.HasParseError()) {
217  throw JsonPointer::Exception("Invalid JSON");
218  }
219 
220  if(!(this->pointerFirst.IsValid())) {
221  throw JsonPointer::Exception("Invalid JSONPointer");
222  }
223 
224  // get result
225  const auto * match{this->pointerFirst.Get(doc)};
226 
227  // check whether match exists
228  if(match != nullptr) {
229  // check type of result
230  if(match->IsString()) {
231  resultTo = std::string(match->GetString(), match->GetStringLength());
232  }
233  else if(match->IsArray() && !(this->textOnly)) {
234  const auto& iterator{match->GetArray().Begin()};
235 
236  if(iterator->IsString()) {
237  resultTo = std::string(iterator->GetString(), iterator->GetStringLength());
238  }
239  else {
240  resultTo = Helper::Json::stringify(*iterator);
241  }
242  }
243  else {
244  // stringify result
245  resultTo = Helper::Json::stringify(*match);
246  }
247  }
248  }

273  inline void JsonPointer::getAll(const rapidjson::Document& doc, std::vector<std::string>& resultTo) const {
274  // empty result
275  resultTo.clear();
276 
277  // check document and pointer
278  if(doc.HasParseError()) {
279  throw JsonPointer::Exception("Invalid JSON");
280  }
281 
282  if(!(this->pointerFirst.IsValid())) {
283  throw JsonPointer::Exception("Invalid JSONPointer");
284  }
285 
286  // check whether multiple matches are possible
287  if(this->pointerStringMulti.empty()) {
288  // get first match only, because multiple matches are not possible
289  const auto * match{this->pointerFirst.Get(doc)};
290 
291  // check whether match exists
292  if(match != nullptr) {
293  // check for array
294  if(match->IsArray() && !(this->textOnly)) {
295  // reserve memory for array members
296  resultTo.reserve(match->GetArray().Size());
297 
298  // go through all array members
299  for(const auto& member : match->GetArray()) {
300  // check for string
301  if(member.IsString()) {
302  resultTo.emplace_back(member.GetString(), member.GetStringLength());
303  }
304  else {
305  // stringify array member
306  resultTo.emplace_back(Helper::Json::stringify(member));
307  }
308  }
309  }
310  // check for string
311  else if(match->IsString()) {
312  resultTo.emplace_back(match->GetString(), match->GetStringLength());
313  }
314  else {
315  // stringify match
316  resultTo.emplace_back(Helper::Json::stringify(*match));
317  }
318  }
319  }
320  else {
321  // get all matches
322  std::size_t counter{0};
323 
324  while(true) {
325  std::string pointerString(this->pointerStringMulti);
326 
327  Helper::Strings::replaceAll(pointerString, "$$", std::to_string(counter));
328 
329  const rapidjson::Pointer pointer{pointerString};
330 
331  if(!(pointer.IsValid())) {
332  throw JsonPointer::Exception("Invalid JSONPointer \'" + pointerString + "\'");
333  }
334 
335  const auto * match{pointer.Get(doc)};
336 
337  if(match == nullptr) {
338  break;
339  }
340 
341  // check type of result
342  if(match->IsString()) {
343  resultTo.emplace_back(match->GetString(), match->GetStringLength());
344  }
345  else {
346  // stringify result
347  resultTo.emplace_back(Helper::Json::stringify(*match));
348  }
349 
350  ++counter;
351  }
352  }
353  }

380  inline void JsonPointer::getSubSets(const rapidjson::Document& doc, std::vector<rapidjson::Document>& resultTo) const {
381  // empty result
382  resultTo.clear();
383 
384  // check document and pointer
385  if(doc.HasParseError()) {
386  throw JsonPointer::Exception("Invalid JSON");
387  }
388 
389  if(!(this->pointerFirst.IsValid())) {
390  throw JsonPointer::Exception("Invalid JSONPointer");
391  }
392 
393  // check whether multiple matches are possible
394  if(this->pointerStringMulti.empty()) { // get first match only, because multiple matches are not possible
395  // get single match
396  const auto * match{this->pointerFirst.Get(doc)};
397 
398  // check whether match exists
399  if(match != nullptr) {
400  // check whether match is an array (and query is not text-only)
401  if(match->IsArray() && !(this->textOnly)) {
402  // reserve memory for results
403  resultTo.reserve(match->GetArray().Size());
404 
405  // go through all array members
406  for(const auto& member : match->GetArray()) {
407  // create new document for the array member at the end of the results
408  resultTo.emplace_back();
409 
410  // copy the array member to the new document
411  resultTo.back().CopyFrom(member, resultTo.back().GetAllocator());
412  }
413  }
414  else {
415  // create a new document for the match at the end of the results
416  resultTo.emplace_back();
417 
418  // copy the match to the new document
419  resultTo.back().CopyFrom(*match, resultTo.back().GetAllocator());
420  }
421  }
422  }
423  else {
424  // get all matches
425  std::size_t counter{0};
426 
427  // loop through all possible matches
428  while(true) {
429  // copy JSONPointer string for placeholder replacement
430  std::string pointerString{this->pointerStringMulti};
431 
432  // replace placeholders with counter value
433  Helper::Strings::replaceAll(pointerString, "$$", std::to_string(counter));
434 
435  // create (and check) JSONPointer
436  const rapidjson::Pointer pointer{pointerString};
437 
438  if(!(pointer.IsValid())) {
439  throw JsonPointer::Exception(
440  "Invalid JSONPointer \'"
441  + pointerString + "\'"
442  );
443  }
444 
445  // get (and check) match
446  const auto * match{pointer.Get(doc)};
447 
448  if(match == nullptr || match->IsNull()) {
449  break;
450  }
451 
452  // create a new document for the match at the end of the results
453  resultTo.emplace_back();
454 
455  // copy the match to the new document
456  resultTo.back().CopyFrom(*match, resultTo.back().GetAllocator());
457 
458  // increment counter
459  ++counter;
460  }
461  }
462  }
