135  inline std::string stringify(const std::vector<std::string>& vectorToStringify) {
136  // create document as array and get reference to allocator
137  rapidjson::Document document;
138 
139  document.SetArray();
140 
141  rapidjson::Document::AllocatorType& allocator{document.GetAllocator()};
142 
143  // reserve memory for all array elements
144  document.Reserve(vectorToStringify.size(), allocator);
145 
146  // write vector elements as string values to array
147  for(const auto& element : vectorToStringify) {
148  rapidjson::Value stringValue;
149 
150  stringValue.SetString(element, allocator);
151 
152  document.PushBack(stringValue, allocator);
153  }
154 
155  // create string buffer and writer
156  rapidjson::StringBuffer buffer;
157  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
158 
159  // write array to string buffer
160  document.Accept(writer);
161 
162  // return string
163  return std::string(buffer.GetString(), buffer.GetSize());
164  }

224 
225  // reserve memory for all array elements
226  document.Reserve(vectorToStringify.size(), allocator);
227 
228  // go through the vector elements representing the objects in the array
229  for(const auto& element : vectorToStringify) {
230  // create object and reserve memory for all [key, value] pairs
231  rapidjson::Value objectValue;
232 
233  objectValue.SetObject();
234 
235  // go through the sub-vector elements representing the [key, value] pairs in the object
236  for(const auto& pair : element) {
237  // create key
238  rapidjson::Value key;
239 
240  key.SetString(pair.first, allocator);
241 
242  // create value
243  rapidjson::Value value;
244 
245  value.SetString(pair.second, allocator);
246 
247  // add [key, value] pair to object
248  objectValue.AddMember(key, value, allocator);

397  inline std::string cleanCopy(std::string_view json) {
398  if(json.empty()) {
399  return std::string();
400  }
401 
402  std::string result;
403 
404  for(std::size_t n{0}; n < json.length(); ++n) {
405  // ignore control characters
406  if(std::iscntrl(json[n]) != 0) {
407  continue;
408  }
409 
410  // check escape sequences
411  if(json[n] == '\\') {
412  bool validEscapeSequence{false};
413 
414  if(n < json.length() - 1) {
415  switch(std::tolower(json[n + 1])) {
416  // check for escaped backslash
417  case '\\':
418  // do not check the following (escaped) backslash...
419  ++n;
420 
421  // ...but add the ignored backslash to the result
422  result.push_back('\\');
423 
424  validEscapeSequence = true;
425 
426  break;
427 
428  // check for single-digit escape sequence names
429  case 'b':
430  case 'f':
431  case 'n':
432  case 'r':
433  case 't':
434  case '\"':
435  case '\'': /* non-standard lenience */
436  case '/':
437  validEscapeSequence = true;
438 
439  break;
440 
441  // check for escaped Unicode character sequence
442  case 'u':
443  if(n < json.length() - unicodeEscapeLength) {
444  validEscapeSequence =
445  ::isxdigit(json[n + unicodeEscapeDigit1]) != 0
446  && ::isxdigit(json[n + unicodeEscapeDigit2]) != 0
447  && ::isxdigit(json[n + unicodeEscapeDigit3]) != 0
448  && ::isxdigit(json[n + unicodeEscapeDigit4]) != 0
449  && ::isxdigit(json[n + unicodeEscapeDigit5]) != 0;
450  }
451 
452  break;
453 
454  default:
455  break;
456  }
457  }
458 
459  if(!validEscapeSequence) {
460  // simply escape the backslash of an invalid escape sequence
461  result.push_back('\\');
462  }
463 
464  result.push_back('\\');
465  }
466  else {
467  result.push_back(json[n]);
468  }
469  }
470 
471  return result;
472  }

485  inline rapidjson::Document parseRapid(std::string_view json) {
486  // clean input
487  std::string cleanJson(cleanCopy(json));
488 
489  rapidjson::Document doc;
490 
491  doc.Parse(cleanJson);
492 
493  if(doc.HasParseError()) {
494  std::string exceptionStr("Json::parseRapid(): ");
495 
496  exceptionStr += rapidjson::GetParseError_En(doc.GetParseError());
497  exceptionStr += " at \'";
498 
499  if(doc.GetErrorOffset() > numDebugChars) {
500  exceptionStr += cleanJson.substr(doc.GetErrorOffset() - numDebugChars, numDebugChars);
501  }
502  else if(doc.GetErrorOffset() > 0) {
503  exceptionStr += cleanJson.substr(0, doc.GetErrorOffset());
504  }
505 
506  exceptionStr += "[!]";
507 
508  if(cleanJson.size() > doc.GetErrorOffset() + numDebugChars) {
509  exceptionStr += cleanJson.substr(doc.GetErrorOffset(), numDebugChars);
510  }
511  else if(cleanJson.size() > doc.GetErrorOffset()) {
512  exceptionStr += cleanJson.substr(doc.GetErrorOffset());
513  }
514 
515  exceptionStr += "\'";
516 
517  throw Exception(exceptionStr);
518  }
519 
520  return doc;
521  }

568  inline Struct::TextMap parseTextMapJson(std::string_view json) {
569  Struct::TextMap result;
570 
571  if(json.empty()) {
572  return result;
573  }
574 
575  // parse JSON
576  rapidjson::Document document{parseRapid(json)};
577 
578  if(!document.IsArray()) {
579  throw Exception("Json::parseTextMapJson(): Invalid text map (is not an array)");
580  }
581 
582  for(const auto& element : document.GetArray()) {
583  if(!element.IsObject()) {
584  throw Exception(
585  "Json::parseTextMapJson(): Invalid text map (an array element is not an object"
586  );
587  }
588 
589  auto p{element.FindMember("p")};
590  auto l{element.FindMember("l")};
591  auto v{element.FindMember("v")};
592 
593  if(p == element.MemberEnd() || !(p->value.IsUint64())) {
594  throw Exception(
595  "Json::parseTextMapJson(): Invalid text map (could not find valid position)"
596  );
597  }
598 
599  if(l == element.MemberEnd() || !(l->value.IsUint64())) {
600  throw Exception(
601  "Json::parseTextMapJson(): Invalid text map (could not find valid length)"
602  );
603  }
604 
605  if(v == element.MemberEnd() || !(v->value.IsString())) {
606  throw Exception(
607  "Json::parseTextMapJson(): Invalid text map (could not find valid value)"
608  );
609  }
610 
611  result.emplace_back(
612  p->value.GetUint64(),
613  l->value.GetUint64(),
614  std::string(v->value.GetString(), v->value.GetStringLength())
615  );
616  }
617 
618  return result;
619  }
