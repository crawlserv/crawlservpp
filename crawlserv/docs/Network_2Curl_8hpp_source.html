340  inline Curl::Curl(
341  std::string_view cookieDirectory,
342  const NetworkOptions& setNetworkOptions
343  ) : cookieDir(cookieDirectory),
344  networkOptions(setNetworkOptions) {
345  // check pointer to libcurl instance
346  if(!(this->curl.valid())) {
347  throw Curl::Exception("Could not initialize the libcurl library");
348  }
349 
350  // get libcurl version information
351  const auto * versionInfo = curl_version_info(CURLVERSION_NOW);
352 
353  this->features = versionInfo->features;
354  this->version = versionInfo->version_num;
355 
356  // check for SSL support
357  if(!(this->hasFeature(CURL_VERSION_SSL))) { //NOLINT(hicpp-signed-bitwise)
358  throw Curl::Exception("The libcurl library does not support SSL");
359  }
360 
361  // configure libcurl
362  this->setOption(CURLOPT_NOSIGNAL, 1L);
363 
364  // set write function
365  //NOLINTNEXTLINE(cppcoreguidelines-pro-type-vararg, hicpp-vararg)
366  this->curlCode = curl_easy_setopt(
367  this->curl.get(),
368  CURLOPT_WRITEFUNCTION,
369  Curl::writer
370  );
371 
372  checkCode(this->curlCode);
373 
374  // set pointer to instance
375  this->setOption(CURLOPT_WRITEDATA, this);
376  }
