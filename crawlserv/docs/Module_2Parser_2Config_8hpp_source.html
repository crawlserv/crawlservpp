390  inline void Config::checkOptions() {
391  // check for target table
392  if(this->config.generalResultTable.empty()) {
393  throw Exception(
394  "Parser::Config::checkOptions():"
395  " No target table has been specified."
396  );
397  }
398 
399  // check properties of date/time queries
400  const auto completeDateTimes{
401  std::min( // number of complete date/time queries (= min. size of all arrays)
402  this->config.parsingDateTimeQueries.size(),
403  this->config.parsingDateTimeSources.size()
404  )
405  };
406 
407  bool incompleteDateTimes{false};
408 
409  // remove date/time queries or sources that are not used
410  if(this->config.parsingDateTimeQueries.size() > completeDateTimes) {
411  this->config.parsingDateTimeQueries.resize(completeDateTimes);
412 
413  incompleteDateTimes = true;
414  }
415  else if(this->config.parsingDateTimeSources.size() > completeDateTimes) {
416  this->config.parsingDateTimeSources.resize(completeDateTimes);
417 
418  incompleteDateTimes = true;
419  }
420 
421  // warn about incomplete date/time queries
422  if(incompleteDateTimes) {
423  this->warning(
424  "\'datetime.queries\', \'.sources\'"
425  " should have the same number of elements."
426  );
427 
428  this->warning(
429  "Incomplete date/time queries removed from configuration."
430  );
431 
432  incompleteDateTimes = false;
433  }
434 
435  // remove date/time formats that are not used, add empty format where none is specified
436  if(this->config.parsingDateTimeFormats.size() > completeDateTimes) {
437  incompleteDateTimes = true;
438  }
439 
440  this->config.parsingDateTimeFormats.resize(completeDateTimes);
441 
442  // replace empty date/time formats with "%F %T"
443  std::replace_if(
444  this->config.parsingDateTimeFormats.begin(),
445  this->config.parsingDateTimeFormats.end(),
446  [](const auto& str) {
447  return str.empty();
448  },
449  "%F %T"
450  );
451 
452  // remove date/time locales that are not used, add empty locale where none is specified
453  if(this->config.parsingDateTimeLocales.size() > completeDateTimes) {
454  incompleteDateTimes = true;
455  }
456 
457  this->config.parsingDateTimeLocales.resize(completeDateTimes);
458 
459  // warn about unused properties
460  if(incompleteDateTimes) {
461  this->warning("Unused date/time properties removed from configuration.");
462  }
463 
464  // check properties of parsing fields
465  const auto completeFields{
466  std::min({ // number of complete fields (= min. size of all arrays)
467  this->config.parsingFieldNames.size(),
468  this->config.parsingFieldQueries.size(),
469  this->config.parsingFieldSources.size()
470  })
471  };
472 
473  bool incompleteFields{false};
474 
475  // remove names of incomplete parsing fields
476  if(this->config.parsingFieldNames.size() > completeFields) {
477  this->config.parsingFieldNames.resize(completeFields);
478 
479  incompleteFields = true;
480  }
481 
482  // remove queries of incomplete parsing fields
483  if(this->config.parsingFieldQueries.size() > completeFields) {
484  this->config.parsingFieldQueries.resize(completeFields);
485 
486  incompleteFields = true;
487  }
488 
489  // remove sources of incomplete parsing fields
490  if(this->config.parsingFieldSources.size() > completeFields) {
491  this->config.parsingFieldSources.resize(completeFields);
492 
493  incompleteFields = true;
494  }
495 
496  // warn about incomplete parsing fields
497  if(incompleteFields) {
498  this->warning(
499  "\'field.names\', \'.queries\' and \'.sources\'"
500  " should have the same number of elements."
501  );
502 
503  this->warning("Incomplete field(s) removed from configuration.");
504 
505  incompleteFields = false;
506  }
507 
508  // remove date/time formats that are not used, add empty format where none is specified
509  if(this->config.parsingFieldDateTimeFormats.size() > completeFields) {
510  incompleteFields = true;
511  }
512 
513  this->config.parsingFieldDateTimeFormats.resize(completeFields);
514 
515  // remove date/time locales that are not used, add empty locale where none is specified
516  if(this->config.parsingFieldDateTimeLocales.size() > completeFields) {
517  incompleteFields = true;
518  }
519 
520  this->config.parsingFieldDateTimeLocales.resize(completeFields);
521 
522  // remove field delimiters that are not used, add empty delimiter (\0) where none is specified
523  if(this->config.parsingFieldDelimiters.size() > completeFields) {
524  incompleteFields = true;
525  }
526 
527  this->config.parsingFieldDelimiters.resize(completeFields, '\0');
528 
529  // replace all empty field delimiters with '\n'
530  std::replace_if(
531  this->config.parsingFieldDelimiters.begin(),
532  this->config.parsingFieldDelimiters.end(),
533  [](char c) {
534  return c == '\0';
535  },
536  '\n'
537  );
538 
539  // remove 'ignore empty values' properties that are not used, set to 'true' where none is specified
540  if(this->config.parsingFieldIgnoreEmpty.size() > completeFields) {
541  incompleteFields = true;
542  }
543 
544  this->config.parsingFieldIgnoreEmpty.resize(completeFields, true);
545 
546  // remove 'save field as JSON' properties that are not used, set to 'false' where none is specified
547  if(this->config.parsingFieldJSON.size() > completeFields) {
548  incompleteFields = true;
549  }
550 
551  this->config.parsingFieldJSON.resize(completeFields, false);
552 
553  // remove 'tidy text' properties that are not used, set to 'false' where none is specified
554  if(this->config.parsingFieldTidyTexts.size() > completeFields) {
555  incompleteFields = true;
556  }
557 
558  this->config.parsingFieldTidyTexts.resize(completeFields, false);
559 
560  // remove 'warning if empty' properties that are not used, set to 'false' where none is specified
561  if(this->config.parsingFieldWarningsEmpty.size() > completeFields) {
562  incompleteFields = true;
563  }
564 
565  this->config.parsingFieldWarningsEmpty.resize(completeFields, false);
566 
567  // warn about unused properties
568  if(incompleteFields) {
569  this->warning("Unused field properties removed from configuration.");
570  }
571 
572  // check properties of ID queries
573  const auto completeIds{
574  std::min( // number of complete ID queries (= min. size of all arrays)
575  this->config.parsingIdQueries.size(),
576  this->config.parsingIdSources.size()
577  )
578  };
579 
580  bool incompleteIds{false};
581 
582  // remove ID queries or sources that are not used
583  if(this->config.parsingIdQueries.size() > completeIds) {
584  this->config.parsingIdQueries.resize(completeIds);
585 
586  incompleteIds = true;
587  }
588  else if(this->config.parsingIdSources.size() > completeIds) {
589  this->config.parsingIdSources.resize(completeIds);
590 
591  incompleteIds = true;
592  }
593 
594  // warn about incomplete ID queries
595  if(incompleteIds) {
596  this->warning(
597  "\'id.queries\' and \'.sources\'"
598  " should have the same number of elements."
599  );
600 
601  this->warning("Incomplete ID queries removed from configuration.");
602  }
603  }
