202  inline void XPath::getFirst(const Parsing::XML& doc, std::string& resultTo) const {
203  // empty result
204  resultTo.clear();
205 
206  // check query and content
207  if(!(this->compiled)) {
208  throw XPath::Exception("No query compiled");
209  }
210 
211  if(!(doc.doc)) {
212  throw XPath::Exception("No content parsed");
213  }
214 
215  // evaluate query with string result
216  try {
217  if(this->query.return_type() == pugi::xpath_type_node_set) {
218  const pugi::xpath_node_set nodeSet(this->query.evaluate_node_set(*(doc.doc)));
219 
220  if(nodeSet.empty()) {
221  resultTo = "";
222  }
223  else {
224  resultTo = XPath::nodeToString(nodeSet[0], this->isTextOnly);
225  }
226  }
227  else {
228  resultTo = this->query.evaluate_string(*(doc.doc));
229  }

232  throw XPath::Exception(e.what());

251  inline void XPath::getAll(const Parsing::XML& doc, std::vector<std::string>& resultTo) const {
252  // empty result
253  resultTo.clear();
254 
255  // check query and content
256  if(!(this->compiled)) {
257  throw XPath::Exception("No query compiled");
258  }
259 
260  if(!(doc.doc)) {
261  throw XPath::Exception("No content parsed");
262  }
263 
264  // evaluate query with multiple string results
265  try {
266  if(this->query.return_type() == pugi::xpath_type_node_set) {
267  const pugi::xpath_node_set nodeSet(
268  this->query.evaluate_node_set(*(doc.doc))
269  );
270 
271  resultTo.reserve(nodeSet.size());
272 
273  for(const auto& node : nodeSet) {
274  const std::string result(XPath::nodeToString(node, this->isTextOnly));
275 
276  if(!result.empty()) {

283

351  // static helper function: convert node to string
352  inline std::string XPath::nodeToString(const pugi::xpath_node& node, bool textOnly) {
353  std::string result;
354 
355  if(node.attribute() != nullptr) {
356  result = node.attribute().as_string();
357  }
358  else if(node.node() != nullptr) {
359  if(textOnly) {
360  XPath::TextOnlyWalker walker;
361 
362  node.node().traverse(walker);
363 
364  result = walker.getResult();
365 
366  if(!result.empty()) {
367  result.pop_back();
368  }
369  }
370  else {
371  for(const auto& child : node.node().children()) {
372  std::ostringstream outStrStr;
373  std::string out;
374 
375  child.print(outStrStr, "", 0);
376 
377  out = outStrStr.str();
378 
379  // parse CDATA
380  if(
381  out.length() > cDataHead.length() + cDataTail.length()
382  && out.substr(0, cDataHead.length()) == cDataHead
383  && out.substr(out.length() - cDataTail.length()) == cDataTail
384  ) {
385  out = out.substr(
386  cDataHead.length(),
387  out.length() - cDataHead.length() - cDataTail.length()
388  );
389  }
390 
391  result += out;
392  }
393  }
394  }
395 
396  return result;
397  }
