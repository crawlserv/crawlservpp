278  inline bool TorControl::newIdentity() {
279  // check whether a TOR control server/port has been set
280  if(!(this->isActive)) {
281  throw Exception("No TOR control server/port set");
282  }
283 
284  // check whether a sufficient amount of time has passed since the last request
285  if(this->newIdentityNotBefore > 0) {
286  this->elapsedMin += this->minTimer.tick();
287 
288  if(this->elapsedMin / millisecondsPerSecond < this->newIdentityNotBefore) {
289  return false;
290  }
291 
292  this->elapsedMin = 0;
293  }
294 
295  // create asio resolver
296  asio::ip::tcp::resolver resolver(this->context);
297 
298  try {
299  // resolve the address of the control server
300  asio::ip::tcp::resolver::results_type endpoints{
301  resolver.resolve(
302  this->server,
303  std::to_string(this->port),
304  asio::ip::tcp::resolver::numeric_service
305  )
306  };
307 
308  // connect to control server
309  asio::connect(this->socket, endpoints);
310 
311  // send authentification
312  const std::string auth("AUTHENTICATE \"" + this->password + "\"\n");
313 
314  asio::write(this->socket, asio::buffer(auth.data(), auth.size()));
315 
316  // read response code (response should be "250 OK" or "515 Bad authentication")
317  std::array<char, responseCodeLength> response{};
318 
319  this->socket.read_some(asio::mutable_buffer(response.data(), responseCodeLength));
320 
321  // check response code
322  if(response[0] != '2' || response[1] != '5' || response[2] != '0') {
323  throw Exception("Authentification failed");
324  }
325 
326  // send command to request a new identity
327  const std::string command("SIGNAL NEWNYM\r\n");
328 
329  asio::write(this->socket, asio::buffer(command.data(), command.size()));
330 
331  // reset timer if necessary
332  if(this->newIdentityAfter > 0) {
333  this->elapsedMax = 0;
334 
335  this->maxTimer.tick();
336  }
337  }
338  catch(const asio::system_error& e) {
339  throw Exception(e.what());
340  }
341 
342  return true;
343  }

350  inline void TorControl::tick() {
351  // check whether timer is enabled
352  if(this->isActive && this->newIdentityAfter > 0) {
353  // get elapsed time (in ms)
354  this->elapsedMin += this->minTimer.tick();
355  this->elapsedMax += this->maxTimer.tick();
356 
357  // check elapsed time (in s)
358  if(this->elapsedMax / millisecondsPerSecond > this->newIdentityAfter) {
359  // request new identity
360  this->newIdentity();
361 
362  // reset timer
363  this->elapsedMax = 0;
364 
365  this->maxTimer.tick();
366  }
367  }
